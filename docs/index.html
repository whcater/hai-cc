import asyncio

async def async_fibonacci(n):
    """
    Asynchronously calculates the nth Fibonacci number.
    (This is for demonstration, fibonacci is CPU-bound and not a good candidate for asyncio)
    """
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    
    a, b = 0, 1
    for _ in range(n - 1):
        a, b = b, a + b
        await asyncio.sleep(0) # Yield control
    return b

# Example usage:
